şu an şöyle bir sıkıntı var ben değer cd a yazıp a içine geidersem sonra cd b yazdıp b içine gidersem ve cd - yazarsam a içine geri dönüyor evet ama b içinde cd blabla gibi olmayan bir dizin adı yazrsam hata döndürüyor ve update_oldpwd içinde o anki dizini güncellediği için b dizinini alıyor ve biçinde kaldığım için cd - basınca a'ya değil sürekli b'ye gidiyor ve loopa giriyor benim dosya yolunu bulamazsam oldpwd'yi güncellememem lazım ki eskisine geri dönebileyim şu an kodlar şöyle: static int	update_oldpwd(t_env **env_list)
{
	char	cwd[1024];

	if (!getcwd(cwd, sizeof(cwd)))
		return (err_exp("OLDPWD: ", 0, 1, 1));
	insert_sorted_exp(env_list, "OLDPWD", cwd);
	return (0);
}

int	cd_cmd(t_command *cmd, t_env **env_list, char *path)
{
	if (cmd->av[1] && cmd->av[2])
		return (err_exp("cd: ", ERR_2_ARG, 0, 1));
	if (!cmd->av[1] || !ft_strcmp(cmd->av[1], "~"))
	{
		update_oldpwd(env_list);
		path = get_env_value(*env_list, "HOME");
		if (!path)
			return (err_exp("cd: ", ERR_HOME, 0, 1));
	}
	else if (!ft_strcmp(cmd->av[1], "-"))
	{
		path = get_env_value(*env_list, "OLDPWD");
		if (!path)
			return (err_exp("cd: ", ERR_OLDPWD, 0, 1));
		printf("%s\n", path);
		update_oldpwd(env_list);
	}
	else if (!update_oldpwd(env_list))
		path = cmd->av[1];
	if ((chdir(path) && cmd->av[1][0]))
		return (err_exp("cd: ", path, 1, 1));
	return (reset_pwd(env_list));
}
